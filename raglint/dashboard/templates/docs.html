ðŸ’¡ Basic Usage

Python
from raglint import RAGPipelineAnalyzer

# Initialize analyzer
analyzer = RAGPipelineAnalyzer(use_smart_metrics=True)

# Prepare your data
data = [
    {
        "query": "What is RAG?",
        "retrieved_contexts": ["RAG combines retrieval and generation..."],
        "response": "RAG is a technique that..."
    }
]

# Run analysis
results = analyzer.analyze(data)

# Get metrics
print(results[0]["metrics"])
